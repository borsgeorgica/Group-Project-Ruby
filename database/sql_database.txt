CREATE TABLE user(

TwitterUsername TEXT NOT NULL,

Name TEXT,
Surname TEXT,

Password TEXT NOT NULL,

Email TEXT NOT NULL,

ContactNumber TEXT NOT NULL,

Address TEXT NOT NULL,

UNIQUE(Email,ContactNumber),

PRIMARY KEY(TwitterUsername)
);


CREATE TABLE twitterOrders(
Username TEXT NOT NULL,
Tweet TEXT NOT NULL,
Time TIMESTAMP NOT NULL,
FOREIGN KEY(Username) REFERENCES user(TwitterUsername)
);

CREATE TABLE twitterFeedback(
Username TEXT NOT NULL,
Tweet TEXT NOT NULL,
Time TIMESTAMP NOT NULL,
FOREIGN KEY(Username) REFERENCES user(TwitterUsername)
);
